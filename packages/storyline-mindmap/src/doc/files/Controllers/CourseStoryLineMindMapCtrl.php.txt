/**
 * Class CourseStoryLineMindMapCtrl
 * @package EONConsulting\Storyline\MindMap\Controllers
 */
class CourseStoryLineMindMapCtrl extends Controller
{
    /**
     * @param Course $course
     * @return mixed
     * @api
     * Returns a Formated JSON for Use with GO JS or HTML LIST
     */

    }

}
